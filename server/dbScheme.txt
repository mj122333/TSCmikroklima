
CREATE TABLE `CVOR` (
	`ID` INT NOT NULL AUTO_INCREMENT,
	`MAC` VARCHAR(12),
	`AKTIVNO` BOOLEAN,
	PRIMARY KEY (`ID`)
);


CREATE TABLE `TEMP` (
	`ID` INT NOT NULL AUTO_INCREMENT,
	`ID_SENZOR` INT NOT NULL,
	`VRIJEDNOST` FLOAT NOT NULL,
	`VRIJEME` TIMESTAMP NOT NULL ON UPDATE CURRENT_TIMESTAMP,
	PRIMARY KEY (`ID`)
);

CREATE TABLE `TEMP_SENZOR` (
	`ID` INT NOT NULL AUTO_INCREMENT,
	`ID_CVOR` INT NOT NULL,
	`ADRESA` VARCHAR(8) NOT NULL,
	`TIP` INT NOT NULL,
	PRIMARY KEY (`ID`)
);


CREATE TABLE `STATUSOBJEKT` (
	`ID` INT NOT NULL AUTO_INCREMENT,
	`PIN` INT NOT NULL,
	`OTVORENI` BOOLEAN NOT NULL,
	`VRIJEME` TIMESTAMP NOT NULL ON UPDATE CURRENT_TIMESTAMP,
	PRIMARY KEY (`ID`)
);

CREATE TABLE `STATUSOBJEKT_SENZOR` (
	`ID` INT NOT NULL AUTO_INCREMENT,
	`ID_CVOR` INT NOT NULL,
	`PIN` VARCHAR(8) NOT NULL,
	`TIP` TEXT(16) CHARACTER SET utf8 COLLATE utf8_bin NOT NULL,
	PRIMARY KEY (`ID`)
);



CREATE TABLE `PROSTORIJA` (
	`ID` INT NOT NULL AUTO_INCREMENT,
	`ID_CVOR` INT NOT NULL,
	`NAZIV` TEXT(64) CHARACTER SET utf8 COLLATE utf8_bin NOT NULL,
	PRIMARY KEY (`ID`)
);
